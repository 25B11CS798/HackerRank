// Function to modify the two integers in place
void modify(int *a, int *b) {
    int original_a = *a;
    int original_b = *b;

    *a = original_a + original_b;       // first number = sum
    *b = abs(original_a - original_b);  // second number = absolute difference
}





int main() {
    int a, b;
    
    // Read input
    scanf("%d", &a);
    scanf("%d", &b);
    
    // Modify values
    modify(&a, &b);
    
    // Print output
    printf("%d\n%d\n", a, b);
    
    return 0;
}

#include <stdio.h>

int main() {
    int n, k;
    scanf("%d %d", &n, &k);

    int max_and = 0;
    int max_or = 0;
    int max_xor = 0;

    for (int a = 1; a <= n; a++) {
        for (int b = a + 1; b <= n; b++) {
            int aand = a & b;
            int aor  = a | b;
            int axor = a ^ b;

            if (aand < k && aand > max_and)
                max_and = aand;

            if (aor < k && aor > max_or)
                max_or = aor;

            if (axor < k && axor > max_xor)
                max_xor = axor;
        }
    }

    printf("%d\n%d\n%d\n", max_and, max_or, max_xor);

    return 0;
}

            int top = i;

/*
 * This stores the total number of books in each shelf.
 */
int* total_number_of_books;

/*
 * This stores the total number of pages in each book of each shelf.
 * The rows represent the shelves and the columns represent the books.
 */
int** total_number_of_pages;
int main() {

    int total_number_of_shelves;
    scanf("%d", &total_number_of_shelves);

    int total_number_of_queries;
    scanf("%d", &total_number_of_queries);

    total_number_of_books = calloc(total_number_of_shelves, sizeof(int));
    total_number_of_pages = malloc(total_number_of_shelves * sizeof(int*));

    for(int i = 0; i < total_number_of_shelves; i++){
        total_number_of_pages[i] = malloc(1100 * sizeof(int)); // max 1100 books per shelf
    }

    while (total_number_of_queries--) {
        
        int type_of_query;
        scanf("%d", &type_of_query);

        if (type_of_query == 1) {
            int x, y;
            scanf("%d %d", &x, &y);

            int index = total_number_of_books[x];           // next empty slot
            total_number_of_pages[x][index] = y;            // store pages
            total_number_of_books[x]++;                     // update count
            // -------------------

        } else if (type_of_query == 2) {
            int x, y;
            scanf("%d %d", &x, &y);
            printf("%d\n", *(*(total_number_of_pages + x) + y));
        } else {
            int x;
            scanf("%d", &x);
            printf("%d\n", *(total_number_of_books + x));
        }
    }

    if (total_number_of_books) {
        free(total_number_of_books);
    }
    
    for (int i = 0; i < total_number_of_shelves; i++) {
        if (*(total_number_of_pages + i)) {
            free(*(total_number_of_pages + i));
        }
    }
    
    if (total_number_of_pages) {
        free(total_number_of_pages);
    }
    
    return 0;
}

int marks_summation(int *marks, int number_of_students, char gender) {
    int sum = 0;

}

void swap(char **a, char **b) {
    char *tmp = *a;
    *a = *b;
    *b = tmp;
}

// Reverse a subarray of strings from start to end
void reverse(char **arr, int start, int end) {
    while (start < end) {
        swap(&arr[start], &arr[end]);
        start++;
        end--;
    }
}

// Generate the next lexicographical permutation
int next_permutation(int n, char **arr) {
    int i = n - 2;
    while (i >= 0 && strcmp(arr[i], arr[i + 1]) >= 0)
        i--;
    if (i < 0) return 0; // last permutation reached

    int j = n - 1;
    while (strcmp(arr[j], arr[i]) <= 0)
        j--;

    swap(&arr[i], &arr[j]);
    reverse(arr, i + 1, n - 1);
    return 1;
}

}

#include <stdio.h>
#include <stdlib.h>
#include <math.h>

struct triangle
{
	int a;
	int b;
	int c;
};

typedef struct triangle triangle;
double heron(int a, int b, int c) {
    double s = (a + b + c) / 2.0;
    return sqrt(s * (s - a) * (s - b) * (s - c));
}

void sort_by_area(triangle* tr, int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (heron(tr[j].a, tr[j].b, tr[j].c) > heron(tr[j + 1].a, tr[j + 1].b, tr[j + 1].c)) {
                triangle temp = tr[j];
                tr[j] = tr[j + 1];
                tr[j + 1] = temp;
            }
        }
    }
}

int main()
{
	int n;
	scanf("%d", &n);
	triangle *tr = malloc(n * sizeof(triangle));
	for (int i = 0; i < n; i++) {
		scanf("%d%d%d", &tr[i].a, &tr[i].b, &tr[i].c);
	}
	sort_by_area(tr, n);
	for (int i = 0; i < n; i++) {
		printf("%d %d %d\n", tr[i].a, tr[i].b, tr[i].c);
	}
	return 0;
}






#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct package {
    char id[100];
    int weight;
} package;

typedef struct post_office {
    int min_weight, max_weight;
    int packages_count;
    package* packages;
} post_office;

typedef struct town {
    char name[100];
    int offices_count;
    post_office* offices;
} town;

// PRINT ALL PACKAGES OF A TOWN
void print_all_packages(town t) {
    printf("%s:", t.name);
    for (int i = 0; i < t.offices_count; i++) {
        printf(" %d:", i);
        for (int j = 0; j < t.offices[i].packages_count; j++) {
            printf(" %s", t.offices[i].packages[j].id);
        }
    }
    printf("\n");
}

// SEND PACKAGES BETWEEN TWO POST OFFICES
void send_all_acceptable_packages(town* t1, int o1, town* t2, int o2) {
    post_office* src = &t1->offices[o1];
    post_office* dest = &t2->offices[o2];

    package* returned = malloc(src->packages_count * sizeof(package));
    int rCount = 0;

    for (int i = 0; i < src->packages_count; i++) {
        package p = src->packages[i];
        if (p.weight >= dest->min_weight && p.weight <= dest->max_weight) {
            dest->packages = realloc(dest->packages, (dest->packages_count + 1) * sizeof(package));
            dest->packages[dest->packages_count++] = p;
        } else {
            returned[rCount++] = p;
        }
    }

    src->packages = realloc(src->packages, rCount * sizeof(package));
    src->packages_count = rCount;
    for (int i = 0; i < rCount; i++)
        src->packages[i] = returned[i];

    free(returned);
}

// FIND TOWN WITH MOST TOTAL PACKAGES
town* town_with_most_packages(town* towns, int n) {
    int max = -1;
    town* res = &towns[0];
    for (int i = 0; i < n; i++) {
        int total = 0;
        for (int j = 0; j < towns[i].offices_count; j++)
            total += towns[i].offices[j].packages_count;
        if (total > max) {
            max = total;
            res = &towns[i];
        }
    }
    return res;
}

// FIND TOWN BY NAME
town* find_town(town* towns, int n, char* name) {
    for (int i = 0; i < n; i++)
        if (strcmp(towns[i].name, name) == 0)
            return &towns[i];
    return NULL;
}

int main() {
    int n;
    scanf("%d", &n);

    town* towns = malloc(n * sizeof(town));

    for (int i = 0; i < n; i++) {
        scanf("%s", towns[i].name);
        scanf("%d", &towns[i].offices_count);

        towns[i].offices = malloc(towns[i].offices_count * sizeof(post_office));

        for (int j = 0; j < towns[i].offices_count; j++) {
            int k;
            scanf("%d %d %d", &k, &towns[i].offices[j].min_weight, &towns[i].offices[j].max_weight);
            towns[i].offices[j].packages_count = k;
            towns[i].offices[j].packages = malloc(k * sizeof(package));

            for (int p = 0; p < k; p++) {
                scanf("%s %d", towns[i].offices[j].packages[p].id, &towns[i].offices[j].packages[p].weight);
            }
        }
    }

    int q;
    scanf("%d", &q);

    while (q--) {
        int type;
        scanf("%d", &type);

        if (type == 1) {
            char name[100];
            scanf("%s", name);
            print_all_packages(*find_town(towns, n, name));
        } else if (type == 2) {
            char t1[100], t2[100];
            int o1, o2;
            scanf("%s %d %s %d", t1, &o1, t2, &o2);
            send_all_acceptable_packages(find_town(towns, n, t1), o1, find_town(towns, n, t2), o2);
        } else if (type == 3) {
            town* res = town_with_most_packages(towns, n);
            printf("Town with the most number of packages is %s\n", res->name);
        }
    }

    // Free memory
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < towns[i].offices_count; j++)
            free(towns[i].offices[j].packages);
        free(towns[i].offices);
    }
    free(towns);

    return 0;
}
